#-------------------------------------------------------
# POST INSTALL CONFIGURATION

%post --log=/var/log/kickstart.log
#!/bin/bash
exec 1> >(tee -a /var/log/kickstart.log)
exec 2>&1

echo "Starting post-install hardening..."

# Enable EPEL
dnf install -y epel-release

#-------------------------------------------------------
# CHRONY HARDENING
#-------------------------------------------------------
cat > /etc/chrony.conf <<EOF
server 0.nl.pool.ntp.org iburst
server 1.nl.pool.ntp.org iburst
driftfile /var/lib/chrony/drift
makestep 1.0 3
rtcsync
bindcmdaddress 127.0.0.1
cmdallow 127.0.0.1
EOF

#-------------------------------------------------------
# LOAD AUDIT RULES
#-------------------------------------------------------
augenrules --load

#-------------------------------------------------------
# SSH HARDENING
#-------------------------------------------------------
cp /etc/ssh/sshd_config /etc/ssh/sshd_config.backup
cat >> /etc/ssh/sshd_config <<EOF
PermitRootLogin no
MaxAuthTries 3
PasswordAuthentication no
AllowAgentForwarding no
AllowTcpForwarding no
X11Forwarding no
ClientAliveInterval 300
ClientAliveCountMax 0
EOF

#-------------------------------------------------------
# DISABLE USB STORAGE (legacy)
#-------------------------------------------------------
echo "install usb-storage /bin/false" > /etc/modprobe.d/usb-storage.conf

#-------------------------------------------------------
# AIDE FILE INTEGRITY
#-------------------------------------------------------
aide --init
mv /var/lib/aide/aide.db.new.gz /var/lib/aide/aide.db.gz
echo "0 3 * * * root /usr/sbin/aide --check" > /etc/cron.d/aide

#-------------------------------------------------------
# IMMUTABLE AUDIT LOGS
#-------------------------------------------------------
chattr +a /var/log/audit/audit.log

#-------------------------------------------------------
# KERNEL HARDENING
#-------------------------------------------------------
cat > /etc/sysctl.d/99-security.conf <<EOF
net.ipv4.ip_forward=0
net.ipv4.conf.all.accept_redirects=0
net.ipv4.conf.all.send_redirects=0
net.ipv4.conf.all.accept_source_route=0
net.ipv4.tcp_syncookies=1
net.ipv4.icmp_echo_ignore_broadcasts=1
net.ipv4.conf.all.log_martians=1
net.ipv4.conf.default.rp_filter=1

kernel.randomize_va_space=2
kernel.kptr_restrict=2
kernel.dmesg_restrict=1

fs.suid_dumpable=0
EOF

sysctl --system

#-------------------------------------------------------
# FINAL PERMISSIONS
#-------------------------------------------------------
chmod 0644 /etc/passwd
chmod 0644 /etc/group
chmod 640 /etc/shadow
chmod 640 /etc/gshadow
chown root:shadow /etc/shadow
chown root:shadow /etc/gshadow
chmod 0600 /etc/ssh/sshd_config

#-------------------------------------------------------
# USBGUARD
#-------------------------------------------------------
systemctl enable usbguard
systemctl start usbguard

#-------------------------------------------------------
# OSQUERY
#-------------------------------------------------------
systemctl enable osqueryd
systemctl start osqueryd

#-------------------------------------------------------
# ZABBIX AGENT 2
#-------------------------------------------------------
systemctl enable zabbix-agent2
systemctl start zabbix-agent2

#-------------------------------------------------------
# WAZUH READINESS (firewall)
#-------------------------------------------------------
firewall-cmd --permanent --add-port=1514/tcp
firewall-cmd --permanent --add-port=1515/tcp
firewall-cmd --reload

#-------------------------------------------------------
# ENABLE SERVICES
#-------------------------------------------------------
systemctl enable auditd chronyd firewalld cockpit.socket fapolicyd

echo "Kickstart hardening complete"
%end

reboot
